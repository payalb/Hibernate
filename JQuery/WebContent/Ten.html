By default case sensitive value selector
  1. equal selector => [name="value"]
  2. not equal selector => [name!="value"]
  3. not equal selector => div:not([name="value"])
  4. contains selector => [name*="value"]
  5. contains word selector, delimited by spaces => [name~="value"]
  6. contains prefix selector (has this value or starts with value followed by hyphen) => [name|="value"]
  7. starts with selector => [name^="value"]
  8. end with selector => [name$="value"]
